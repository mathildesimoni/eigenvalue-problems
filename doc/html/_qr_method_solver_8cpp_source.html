12void QrMethodSolver<T>::QrDecomposition(const Matrix<T> &A_iter, Matrix<T> &Q, Matrix<T> &R)
13{
14 // A_shifted is passed as a const so that it can not be modified
15
16 int n = A_iter.rows(); // Number of rows (same as number of columns)
17
18 // Initialize Q and R
19 Q = Matrix<T>::Identity(n, n);
20 R = A_iter;
21
22 for (int k = 0; k < n; k++)
23 {
24
25 Vector<T> x = R.block(k, k, n - k, 1);
26 Vector<T> v = x;
27 v(0) += (x.norm() * (x(0) >= 0 ? 1 : -1)); // v = x + sign(x_0) * ||x|| * e_1
28
29 v = v / v.norm(); // Normalize the vector v
30
31 // Apply Householder transformation to the trailing matrix
32 R.block(k, k, n - k, n - k) -= 2 * v * v.transpose() * R.block(k, k, n - k, n - k);
33
34 // Apply Householder transformation to Q
35 Q.block(0, k, n, n - k) -= 2 * Q.block(0, k, n, n - k) * v * v.transpose();
36 }
37}

64 A_iter = R * Q;
65
66 // Compute the error as the sum of the absolute below-diagonal elements

73 }
74 }
